(function() {
  var Classy,
    slice = [].slice;

  Classy = require('../specs/Classy');
  /**
   * Classy.Maths
   * @param {args...} Multiple arguments supports
   * @plugin {Math Library and much more} supports the built-in JS/CommonJS math standard library
   * @returns {eval}
   */
  Classy.Maths = (function(_this) {
    return function() {
      var args;
      args = 1 <= arguments.length ? slice.call(arguments, 0) : [];
      return eval.apply(null, args);
    };
  })(this);

  module.exports = Classy.Maths;

}).call(this);
