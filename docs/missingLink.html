(function() {
  var Classy, _;

  _ = require('underscore-plus');

  Classy = require('../specs/Classy');
  /**
   * Classy.ClassCut
   * @param {_class} CSS either #id or .class
   * @param {opts} Optional Options or interal code
   * @param {from} Integer to start the loop
   * @param {to} Integer to end the loop
   * @param {_by} Integer to range the `from` and `to` Integer
   * @param {$scope} The base integer to cut the from and to integer
   * @using {Underscore.js}
   */
  Classy.ClassCut = function(_class, opts, from, to, _by) {
    var $idem, $scope, i, len, template;
    $scope = _.range(from, to, _by);
    for (i = 0, len = $scope.length; i < len; i++) {
      $idem = $scope[i];
      template = _class + "-" + $idem + " {\n	" + opts + "\n}";
    }
    return template;
  };

  module.exports = Classy.ClassCut;

}).call(this);
