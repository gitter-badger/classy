 * unquote.js
 * Unhand the string and get back your text!
 *
 * Author: Nicky Laczko

(function() {
  var Classy, _;

  Classy = require('../specs/Classy');

  _ = require('underscore-plus');

  Classy.unquote = function(str) {
    if (typeof str === "String") {
      return _.escape(str);
    }
  };

  module.exports = Classy.unquote;

}).call(this);
